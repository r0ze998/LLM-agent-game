AI: 3層意思決定エンジン

P2
本能 (Instinct) — ルールベース即応
hunger < 20 → eat / energy < 15 → sleep
P1
日課計画 (Daily Plan) — LLM 1日1回生成
24スロットのスケジュール + socialIntentions
P0
探索 (Fallback) — 計画外の状況
新規状況に対応 / 世界を探索 
// decisionEngine.ts — decide()
async function decide(ctx: DecisionContext): Promise<AgentAction> {
  // P2: Instinct — 生存本能が最優先
  const instinct = checkInstincts(agent);
  if (instinct) return instinct;

  // P1: Daily Plan — LLMが朝に生成した計画に従う
  if (tickOfDay === 0 || !plan)
    plan = await generateDailyPlan(ctx);
  const scheduled = getScheduledAction(plan, tickOfDay);
  if (scheduled) return scheduled;

  // P0: Fallback — 探索行動
  return { type: 'explore' };
}

Cairo: 物理法則の不変性

// physics.cairo — Layer 0: clamp_effect
// 全Effectは物理法則の範囲内に強制クランプされる
fn clamp_effect(
    config: @PhysicsConfig,
    effect_type: EffectType,
    value: i128
) -> i128 {
    let (lo, hi) = Self::get_bounds(config, effect_type);
    if value < lo { lo }
    else if value > hi { hi }
    else { value }
}
